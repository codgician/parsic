Combinator: trim

Consume as many whitespace characters ( , \n, \r or \t) as possible surrounding given parser. trim(p) is equivalant to mid(space().many(), p, space().many()).

Example

use naive_parsec::core::Parsable;
use naive_parsec::primitives::{CharStream, literal, PrimitiveExt, trim};

// Consume a whitespace character
let parser = literal("Hello").trim();

let mut st = CharStream::new("   Hello   ");
let (res, logs) = parser.exec(&mut st);

assert_eq!(Some("Hello"), res);
assert_eq!("", st.as_str());
assert_eq!(0, logs.len());
